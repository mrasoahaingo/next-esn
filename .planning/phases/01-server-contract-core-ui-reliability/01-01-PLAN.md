For `extract-cv.ts` and `positioning-analyze.ts` and `positioning-generate.ts` (candidates/positionings tables -- they have `status` and `workflow_run_id` columns):

```typescript
async function handleWorkflowError(
  recordId: string,
  table: 'candidates' | 'positionings',
  error: unknown,
) {
  'use step';

  const supabase = getSupabase();
  const errorMessage = error instanceof Error ? error.message : 'Unknown error';

  await supabase
    .from(table)
    .update({
      status: 'error',
      workflow_run_id: null,
    })
    .eq('id', recordId);

  // Write error frame to NDJSON stream so connected clients see it
  const writable = getWritable<Uint8Array>();
  const writer = writable.getWriter();
  const encoder = new TextEncoder();
  try {
    await writer.write(
      encoder.encode(JSON.stringify({ error: errorMessage }) + '\n'),
    );
  } finally {
    writer.releaseLock();
    await writable.close();
  }
}
handleWorkflowError.maxRetries = 0;
```
